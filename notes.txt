----------args-----------

# select the required seed number.
    -1 or nothing will increment
    the previous seed by 1:       --seed <seed_number>
checkout specific branch:         --branch <branch_to_use>
run "make init":                  --init
run "make build-solution":        --build
pull the branch:                  --pull
make run-solution-extra-cpus:     --extra-cpus
make run-sim-no-gui               --no-sim-gui

e.g: python3 update_config.py --branch master --pull --init --build --seed 30923

----------scripts-----------
# enter the required seed number. Run only ONE.
python3 update_config.py --branch master --seed 1000
python3 update_config.py --branch master --pull True --seed 1000
